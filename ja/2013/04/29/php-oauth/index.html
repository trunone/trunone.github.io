編譯與安裝 PHP OAuth

% sudo apt-get install php5-dev


% wget http://pecl.php.net/get/oauth
% tar -xvzf oauth-x.x.x.tgz
% cd oauth-x.x.x/
% phpize
% ./configure
% make 
% sudo make install
% sudo echo extension=/usr/lib/php5/[date]/oauth.so > /etc/php5/conf.d/oauth.ini
